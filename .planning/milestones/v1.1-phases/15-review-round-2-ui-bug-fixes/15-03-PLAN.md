```csharp
private void DelayedCollapse(UIElement element)
{
    var timer = new System.Windows.Threading.DispatcherTimer
    {
        Interval = TimeSpan.FromMilliseconds(100)
    };
    timer.Tick += (_, _) =>
    {
        timer.Stop();
        if (element.Opacity == 0)
            element.Visibility = Visibility.Collapsed;
    };
    timer.Start();
}
```

private void ApplyActiveHighlight(ListBoxItem item)
{
    if (item.Content is Border outerBorder)
    {
        outerBorder.Background = GetBrush("c-selected-bg");

        // R2-TAB-03: Show pin/close buttons on selected tab (no hover needed)

                    btn.Visibility = Visibility.Visible;

        }
    }
}
