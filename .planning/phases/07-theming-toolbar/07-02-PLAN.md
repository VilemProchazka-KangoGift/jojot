```csharp
// ─── Toolbar event handlers (Phase 7) ──────────────────────────────

private void ToolbarUndo_Click(object sender, RoutedEventArgs e) => PerformUndo();
private void ToolbarRedo_Click(object sender, RoutedEventArgs e) => PerformRedo();

private void ToolbarPin_Click(object sender, RoutedEventArgs e)
{
    if (_activeTab != null)
        _ = TogglePinAsync(_activeTab);
}

private void ToolbarClone_Click(object sender, RoutedEventArgs e)
{
    if (_activeTab != null)
        _ = CloneTabAsync(_activeTab);
}

private void ToolbarCopy_Click(object sender, RoutedEventArgs e) => CopyContent();

private void ToolbarPaste_Click(object sender, RoutedEventArgs e)
{
    // Focus editor first to ensure paste target, then trigger paste
    ContentEditor.Focus();
    ApplicationCommands.Paste.Execute(null, ContentEditor);
}

private void ToolbarSave_Click(object sender, RoutedEventArgs e) => _ = SaveAsTxtAsync();

private void ToolbarDelete_Click(object sender, RoutedEventArgs e)
{
    if (_activeTab != null)
        _ = DeleteTabAsync(_activeTab);
}
```

**Add toolbar state update method:**

```csharp
private void UpdateToolbarState()
{
    bool hasTab = _activeTab != null;

    ToolbarUndo.IsEnabled = hasTab && UndoManager.CanUndo(_activeTab!.Id);
    ToolbarRedo.IsEnabled = hasTab && UndoManager.CanRedo(_activeTab!.Id);
    ToolbarPin.IsEnabled = hasTab;
    ToolbarClone.IsEnabled = hasTab;
    ToolbarCopy.IsEnabled = hasTab;
    ToolbarPaste.IsEnabled = hasTab;
    ToolbarSave.IsEnabled = hasTab;
    ToolbarDelete.IsEnabled = hasTab;

    // Update pin icon: show unpin icon if tab is already pinned
    if (hasTab && _activeTab!.Pinned)
    {
        PinIconText.Text = "\uE77A";  // Unpin
        ToolbarPin.ToolTip = "Unpin (Ctrl+P)";
    }
    else
    {
        PinIconText.Text = "\uE718";  // Pin
        ToolbarPin.ToolTip = "Pin (Ctrl+P)";
    }
}
```

**Call UpdateToolbarState() from key state change points:**
1. **Tab selection changed:** In `TabList_SelectionChanged` — after updating `_activeTab`, call `UpdateToolbarState()`.
2. **After undo/redo:** In `PerformUndo()` and `PerformRedo()` — call `UpdateToolbarState()` after the operation.
3. **After pin toggle:** After `TogglePinAsync` completes — call `UpdateToolbarState()`.
4. **After text changes:** In the TextChanged handler — call `UpdateToolbarState()` (undo/redo availability changes with edits).
5. **Initial state:** Call `UpdateToolbarState()` at the end of the constructor or after InitializeComponent (all buttons disabled since no tab is selected).

**Delete button hover opacity:** Add MouseEnter/MouseLeave handlers in code-behind (or wire in constructor/Loaded):
```csharp
ToolbarDelete.MouseEnter += (s, e) => DeleteIconText.Opacity = 1.0;
ToolbarDelete.MouseLeave += (s, e) => DeleteIconText.Opacity = 0.7;
```

**Check UndoManager API:** Verify the exact CanUndo/CanRedo method signatures. They should be something like:
- `UndoManager.CanUndo(long tabId)` returns bool
- `UndoManager.CanRedo(long tabId)` returns bool

If UndoManager uses a different API shape (e.g., gets the stack first then checks), adapt accordingly.

**Edge case:** When there's no active tab, ALL toolbar buttons must be disabled. This prevents null reference exceptions from toolbar clicks when the tab list is empty.
