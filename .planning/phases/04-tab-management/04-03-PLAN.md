```csharp
private void BeginRename(ListBoxItem item, NoteTab tab)
{
    if (_activeRename != null) CommitRename();

    // Find the rename TextBox in the item's visual tree
    var renameBox = FindDescendant<TextBox>(item);
    var labelBlock = renameBox.Tag as TextBlock;
    if (renameBox == null || labelBlock == null) return;

    labelBlock.Visibility = Visibility.Collapsed;
    renameBox.Text = tab.Name ?? "";
    renameBox.Visibility = Visibility.Visible;
    renameBox.SelectAll();
    Keyboard.Focus(renameBox);

    _activeRename = (item, tab, renameBox);

    // Wire up events (remove when done)
    renameBox.PreviewKeyDown += RenameBox_PreviewKeyDown;
    renameBox.LostFocus += RenameBox_LostFocus;
}
```

**CommitRename():**
```csharp
private void CommitRename()
{
    if (_activeRename == null) return;
    var (item, tab, box) = _activeRename.Value;

    // Unwire events
    box.PreviewKeyDown -= RenameBox_PreviewKeyDown;
    box.LostFocus -= RenameBox_LostFocus;

    string newName = box.Text.Trim();
    // TABS-07: empty/whitespace clears custom name
    tab.Name = string.IsNullOrWhiteSpace(newName) ? null : newName;

    box.Visibility = Visibility.Collapsed;
    var labelBlock = box.Tag as TextBlock;
    if (labelBlock != null) labelBlock.Visibility = Visibility.Visible;

    UpdateTabItemDisplay(tab);
    _ = DatabaseService.UpdateNoteNameAsync(tab.Id, tab.Name);

    _activeRename = null;
}
```

**CancelRename():**
```csharp
private void CancelRename()
{
    if (_activeRename == null) return;
    var (item, tab, box) = _activeRename.Value;

    box.PreviewKeyDown -= RenameBox_PreviewKeyDown;
    box.LostFocus -= RenameBox_LostFocus;

    box.Visibility = Visibility.Collapsed;
    var labelBlock = box.Tag as TextBlock;
    if (labelBlock != null) labelBlock.Visibility = Visibility.Visible;

    _activeRename = null;
}

private async Task TogglePinAsync(NoteTab tab)
{
    tab.Pinned = !tab.Pinned;
    await DatabaseService.UpdateNotePinnedAsync(tab.Id, tab.Pinned);

    // Re-sort: pinned to top, then by sort_order
    var sorted = _tabs.OrderByDescending(t => t.Pinned).ThenBy(t => t.SortOrder).ToList();
    _tabs.Clear();
    foreach (var t in sorted) _tabs.Add(t);

    // Reassign sort_order to match new positions
    for (int i = 0; i < _tabs.Count; i++)
        _tabs[i].SortOrder = i;
    await DatabaseService.UpdateNoteSortOrdersAsync(_tabs.Select(t => (t.Id, t.SortOrder)));

    RebuildTabList();
    SelectTabByNote(tab);  // Re-select the toggled tab
}
```

**CloneTabAsync(NoteTab source):**
```csharp
private async Task CloneTabAsync(NoteTab source)
{
    // Save current content first
    SaveCurrentTabContent();

    int newSortOrder = source.SortOrder + 1;

    // Shift sort_order of all tabs after the clone position
    foreach (var tab in _tabs.Where(t => t.SortOrder >= newSortOrder))
        tab.SortOrder++;

    long newId = await DatabaseService.InsertNoteAsync(
        _desktopGuid, source.Name, source.Content,
        source.Pinned, newSortOrder);

    var clone = new NoteTab
    {
        Id = newId, DesktopGuid = _desktopGuid,
        Name = source.Name, Content = source.Content,
        Pinned = source.Pinned, SortOrder = newSortOrder,
        CreatedAt = DateTime.Now, UpdatedAt = DateTime.Now
    };

    // Insert into collection at correct position
    int insertIndex = _tabs.IndexOf(source) + 1;
    _tabs.Insert(insertIndex, clone);

    // Persist sort orders
    await DatabaseService.UpdateNoteSortOrdersAsync(_tabs.Select(t => (t.Id, t.SortOrder)));

    RebuildTabList();
    SelectTabByNote(clone);
    Keyboard.Focus(ContentEditor);
}
```

**TabItem_PreviewMouseMove:**
```csharp
private void TabItem_PreviewMouseMove(object sender, MouseEventArgs e)
{
    if (e.LeftButton != MouseButtonState.Pressed || _dragItem == null || _dragTab == null) return;

    Point current = e.GetPosition(TabList);
    Vector diff = _dragStartPoint - current;

    // Minimum 5px distance before dragging starts
    if (Math.Abs(diff.Y) < 5 && Math.Abs(diff.X) < 5) return;

    if (!_isDragging)
    {
        _isDragging = true;
        _dragItem.Opacity = 0.6;
        Mouse.Capture(TabList);
    }

    // Find drop position
    UpdateDropIndicator(current);
}
```

**UpdateDropIndicator(Point mousePos):**
Iterate through TabList.Items to find which gap the mouse is over. The drop indicator is a thin Border (Height=2, Background=accent color) added temporarily to the tab list visual.

For zone enforcement: if _dragTab.Pinned, only allow drop between pinned items. If !_dragTab.Pinned, only allow drop between unpinned items. Skip separator items.

```csharp
private void UpdateDropIndicator(Point mousePos)
{
    // Remove existing indicator
    RemoveDropIndicator();

    int dropIndex = -1;
    double bestDistance = double.MaxValue;

    for (int i = 0; i < TabList.Items.Count; i++)
    {
        if (TabList.Items[i] is not ListBoxItem candidate || candidate.Tag is not NoteTab candidateTab)
            continue;

        // Zone enforcement
        if (candidateTab.Pinned != _dragTab!.Pinned) continue;

        // Get the Y position of this item
        var transform = candidate.TransformToAncestor(TabList);
        Point itemPos = transform.Transform(new Point(0, 0));
        double itemMid = itemPos.Y + candidate.ActualHeight / 2;

        double distance = Math.Abs(mousePos.Y - itemPos.Y);
        if (distance < bestDistance)
        {
            bestDistance = distance;
            dropIndex = i;
        }

        distance = Math.Abs(mousePos.Y - (itemPos.Y + candidate.ActualHeight));
        if (distance < bestDistance)
        {
            bestDistance = distance;
            dropIndex = i + 1;
        }
    }

    _dragInsertIndex = dropIndex;

    // Show visual indicator (a thin colored line)
    // For simplicity, change the border-top of the item at dropIndex
    if (dropIndex >= 0 && dropIndex < TabList.Items.Count)
    {
        if (TabList.Items[dropIndex] is ListBoxItem targetItem && targetItem.Content is Border border)
        {
            border.BorderThickness = new Thickness(border.BorderThickness.Left, 2, 0, 0);
            border.BorderBrush = new SolidColorBrush(Color.FromRgb(0x21, 0x96, 0xF3));
            _dropIndicator = border;
        }
    }
}
```

**CompleteDrag():**
```csharp
private void CompleteDrag()
{
    if (!_isDragging) { ResetDragState(); return; }

    Mouse.Capture(null);
    RemoveDropIndicator();

    if (_dragItem != null) _dragItem.Opacity = 1.0;

    if (_dragInsertIndex >= 0 && _dragTab != null)
    {
        // Calculate the NoteTab index from the ListBoxItem index
        // Move the tab in _tabs collection
        int oldIndex = _tabs.IndexOf(_dragTab);
        int newIndex = CalculateCollectionIndex(_dragInsertIndex);

        if (oldIndex >= 0 && newIndex >= 0 && oldIndex != newIndex)
        {
            _tabs.Move(oldIndex, newIndex);

            // Reassign sort_order to match new collection order
            for (int i = 0; i < _tabs.Count; i++)
                _tabs[i].SortOrder = i;

            _ = DatabaseService.UpdateNoteSortOrdersAsync(
                _tabs.Select(t => (t.Id, t.SortOrder)));

            RebuildTabList();
            SelectTabByNote(_dragTab);
        }
    }

    ResetDragState();
}
```

**CalculateCollectionIndex(int listBoxIndex):**
Maps from ListBoxItem index (which may include separator items) back to the _tabs collection index. Iterate TabList.Items up to listBoxIndex, counting only items whose Tag is NoteTab.

**RemoveDropIndicator():**
```csharp
private void RemoveDropIndicator()
{
    if (_dropIndicator != null)
    {
        // Reset the border we modified
        _dropIndicator.BorderThickness = new Thickness(2, 0, 0, 0);
        // Restore original border brush based on selection state
        var item = _dropIndicator.Parent as ListBoxItem;
        if (item != null && TabList.SelectedItem == item)
            _dropIndicator.BorderBrush = new SolidColorBrush(Color.FromRgb(0x21, 0x96, 0xF3));
        else
            _dropIndicator.BorderBrush = Brushes.Transparent;
        _dropIndicator = null;
    }
}
```

**ResetDragState():**

private void ResetDragState()
{

    _dragItem = null;
    _dragTab = null;
    _dragInsertIndex = -1;
}
